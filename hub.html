async function applyCustomization() {
  if (!custGeneratedCode) return;
  if (!serverMode) { alert('Server not running. Start with: python server.py'); return; }

  const status = document.getElementById('cust-apply-status');
  status.textContent = 'Saving snapshot…'; status.style.color = 'var(--text3)';

  // Save server-side snapshot before applying
  try {
    const site = (cfg.sites||[]).find(s => s.id === custGeneratedSiteId);
    await fetch('/api/snapshot', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ label: `Before customize: ${site?.domain || custGeneratedSiteId}` }),
    });
  } catch {}

  const site = (cfg.sites||[]).find(s => s.id === custGeneratedSiteId);
  if (!site) { status.textContent = 'Site not found.'; return; }

  site[custGeneratedTarget] = (site[custGeneratedTarget] || '') + '\n' + custGeneratedCode;

  await saveConfigToServer();
  status.textContent = '✓ Applied! Publish to GitHub when ready.'; status.style.color = 'var(--green)';
  showCustApplied();
  addLog('ok', 'customize', `Customization applied to ${site.domain} → ${custGeneratedTarget}`);
}

function discardCustomization() {
  custGeneratedCode = '';
  document.getElementById('cust-preview').style.display = 'none';
}

async function generateComments() {
  const siteId = document.getElementById('comm-site').value;
  const slug = document.getElementById('comm-slug').value.trim();
  const title = document.getElementById('comm-title').value.trim();
  const count = Math.max(1, Math.min(19, parseInt(document.getElementById('comm-count').value) || 6));
  const days = Math.max(1, parseInt(document.getElementById('comm-days').value) || 14);

  if (!slug) { alert('Enter the article slug.'); return; }
  const apiKey = ls('anthropicKey');
  if (!apiKey) { alert('Add Anthropic API key in Settings.'); showPanel('settings'); return; }

  const site = (cfg?.sites||[]).find(s => s.id === siteId);
  const btn = document.getElementById('comm-gen-btn');
  const thinking = document.getElementById('comm-thinking');
  btn.style.display = 'none'; thinking.classList.add('show');
  generatedComments = [];
  document.getElementById('comm-push-btn').style.display = 'none';

  const today = new Date().toISOString().slice(0,10);
  const COMM_COLORS = ['#3ecf8e','#38bdf8','#a78bfa','#f59e0b','#ef4444','#10b981','#f97316','#ec4899','#6366f1','#14b8a6'];
  const sysPrompt = `You generate realistic blog comments for a ${site?.category||'finance/investing'} article. Write as distinct real readers: brief, specific, occasionally opinionated. Mix agreement, questions, and brief personal anecdotes. Never use em-dashes. Return a JSON array of exactly ${count} objects with keys: name (FirstName L. format), text (1-3 sentences), date_iso (YYYY-MM-DD, spread across ${days} days starting ${today}), likes (integer 0-12). Return ONLY the JSON array, no markdown.`;

  try {
    const r = await fetch('/api/claude', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        _api_key: apiKey,
        model: 'claude-haiku-4-5-20251001',
        max_tokens: 2000,
        messages: [{ role: 'user', content: `Article: "${title || slug}"\nSite niche: ${site?.category || ''}\nGenerate ${count} comments.` }],
        system: sysPrompt,
      })
    });
    const data = await r.json();
    if (!r.ok) throw new Error(data.error?.message || r.status);
    let text = (data.content?.[0]?.text || '').trim().replace(/^```json\s*/,'').replace(/\s*```$/,'');
    const raw = JSON.parse(text);
    const MONTHS = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
    generatedComments = raw.map((c, i) => {
      const name = c.name || 'Reader';
      const ini = name.split(/\s+/).map(w => w[0]).join('').slice(0,2).toUpperCase();
      const color = COMM_COLORS[(name.charCodeAt(0) + i) % COMM_COLORS.length];
      const dateStr = c.date_iso || c.date || today;
      const d = new Date(dateStr + 'T00:00:00');
      const date_display = MONTHS[d.getMonth()] + ' ' + d.getDate() + ', ' + d.getFullYear();
      return { id: `c${i+1}${Math.floor(Math.random()*900+100)}`, name, initials: ini, color, date_iso: dateStr, date_display, text: c.text || '', likes: c.likes || 0 };
    });
    generatedCommentsSlug = slug;
    generatedCommentsSiteId = siteId;
    renderCommPreview();
    document.getElementById('comm-push-btn').style.display = '';
    addLog('info', 'comments', `Generated ${generatedComments.length} comments for ${slug}`);
  } catch(e) {
    alert('Error: ' + e.message);
    addLog('err', 'comments', 'Comment generation failed: ' + e.message);
  } finally {
    btn.style.display = ''; thinking.classList.remove('show');
  }
}

function renderCommPreview() {
  const list = document.getElementById('comm-preview-list');
  if (!generatedComments.length) { list.textContent = 'No comments generated.'; return; }
  const COLORS = ['#3ecf8e','#38bdf8','#a78bfa','#f59e0b','#ef4444','#10b981'];
  list.innerHTML = generatedComments.map((c, i) => {
    const initials = (c.name||'?').split(' ').map(w=>w[0]).join('').slice(0,2).toUpperCase();
    const color = COLORS[i % COLORS.length];
    return `

      
${escHtml(initials)}

      

        

          ${escHtml(c.name||'')}
          ${c.date_display||c.date||''}
          ♥ ${c.likes||0}
        

        
${escHtml(c.text||'')}

      

    
`;
  }).join('');
}

async function pushGeneratedComments() {
  if (!generatedComments.length) return;
  const token = ls('ghToken');
  const site = (cfg?.sites||[]).find(s => s.id === generatedCommentsSiteId);
  if (!token) { alert('Set GitHub token in Settings.'); return; }
  if (!site?.repo || site.repo.includes('YOUR_')) { alert('Set the site repo in network-config.json first.'); return; }

  const btn = document.getElementById('comm-push-btn');
  btn.textContent = 'Pushing…'; btn.disabled = true;

  const branch = ls('branch') || 'main';
  const path = `${generatedCommentsSlug}/comments.json`;
  const content = JSON.stringify(generatedComments, null, 2);
  const b64 = btoa(unescape(encodeURIComponent(content)));

  try {
    const getR = await fetch(`/api/github/repos/${site.repo}/contents/${path}`, {
      headers: { 'X-GH-Token': token }
    });
    let sha = null;
    if (getR.ok) { const d = await getR.json(); sha = d.sha; }
    const body = { message: `Add comments: ${generatedCommentsSlug}`, content: b64, branch };
    if (sha) body.sha = sha;
    const r = await fetch(`/api/github/repos/${site.repo}/contents/${path}`, {
      method: 'PUT',
      headers: { 'X-GH-Token': token, 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    });
    if (!r.ok) { const e = await r.json(); throw new Error(e.message || r.status); }
    btn.textContent = '✓ Pushed!';
    addLog('ok', 'comments', `Comments pushed: ${site.domain}/${generatedCommentsSlug}/comments.json`);
    setTimeout(() => { btn.textContent = '↑ Push to GitHub'; btn.disabled = false; }, 2500);
  } catch(e) {
    alert('Push failed: ' + e.message);
    btn.textContent = '↑ Push to GitHub'; btn.disabled = false;
  }
}

async function addManualComment() {
  const siteId = document.getElementById('comm-manual-site').value;
  const slug = document.getElementById('comm-manual-slug').value.trim();
  const name = document.getElementById('comm-manual-name').value.trim();
  const text = document.getElementById('comm-manual-text').value.trim();
  const date = document.getElementById('comm-manual-date').value || new Date().toISOString().slice(0,10);
  const status = document.getElementById('comm-manual-status');

  if (!slug || !name || !text) { status.textContent = 'Fill in slug, name, and text.'; status.style.color='var(--red)'; return; }

  const token = ls('ghToken');
  const site = (cfg?.sites||[]).find(s => s.id === siteId);
  if (!token) { alert('Set GitHub token in Settings.'); return; }
  if (!site?.repo || site.repo.includes('YOUR_')) { alert('Set the site repo in config first.'); return; }

  status.textContent = 'Fetching existing comments…'; status.style.color='var(--text3)';
  const branch = ls('branch') || 'main';
  const path = `${slug}/comments.json`;

  try {
    let existing = [];
    let sha = null;
    const getR = await fetch(`/api/github/repos/${site.repo}/contents/${path}`, {
      headers: { 'X-GH-Token': token }
    });
    if (getR.ok) {
      const d = await getR.json();
      sha = d.sha;
      existing = JSON.parse(decodeURIComponent(escape(atob(d.content.replace(/\n/g,'')))));
    }
    const COMM_COLORS_M = ['#3ecf8e','#38bdf8','#a78bfa','#f59e0b','#ef4444','#10b981','#f97316','#ec4899','#6366f1','#14b8a6'];
    const ini = name.split(/\s+/).map(w => w[0]).join('').slice(0,2).toUpperCase();
    const col = COMM_COLORS_M[(name.charCodeAt(0) + existing.length) % COMM_COLORS_M.length];
    const MONTHS_M = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
    const dObj = new Date(date + 'T00:00:00');
    const date_display = MONTHS_M[dObj.getMonth()] + ' ' + dObj.getDate() + ', ' + dObj.getFullYear();
    existing.push({ id: `m${Date.now()}`, name, initials: ini, color: col, date_iso: date, date_display, text, likes: 0 });
    const b64 = btoa(unescape(encodeURIComponent(JSON.stringify(existing, null, 2))));
    const body = { message: `Add comment: ${slug}`, content: b64, branch };
    if (sha) body.sha = sha;
    const r = await fetch(`/api/github/repos/${site.repo}/contents/${path}`, {
      method: 'PUT',
      headers: { 'X-GH-Token': token, 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    });
    if (!r.ok) { const e = await r.json(); throw new Error(e.message || r.status); }
    status.textContent = '✓ Comment added!'; status.style.color='var(--green)';
    ['comm-manual-name','comm-manual-text'].forEach(id => document.getElementById(id).value='');
    addLog('ok', 'comments', `Manual comment added: ${site.domain}/${slug}`);
    setTimeout(() => status.textContent = '', 2500);
  } catch(e) {
    status.textContent = 'Error: ' + e.message; status.style.color='var(--red)';
  }
}

function loadCommSchedule() {
  if (!cfg) return;
  const siteId = document.getElementById('comm-sched-site')?.value;
  const site = (cfg.sites||[]).find(s => s.id === siteId);
  const hint = document.getElementById('comm-sched-hint');
  const fields = document.getElementById('comm-sched-fields');
  if (!site) { if(hint) hint.style.display=''; if(fields) fields.style.display='none'; return; }
  const cs = site.comment_schedule || {};
  document.getElementById('cs-enabled').checked = cs.enabled || false;
  document.getElementById('cs-initial').value = cs.initial_count ?? 6;
  document.getElementById('cs-max').value = cs.max_count ?? 15;
  document.getElementById('cs-perday').value = cs.add_per_day ?? 2;
  document.getElementById('cs-delay').value = cs.delay_days ?? 0;
  if (hint) hint.style.display = 'none';
  if (fields) fields.style.display = '';
}

async function saveCommSchedule() {

  };
  await saveConfigToServer();
  const status = document.getElementById('cs-save-status');
  status.textContent = '✓ Saved'; status.style.color='var(--green)';
  addLog('ok', 'comments', `Comment schedule saved for ${site.domain}`);
  setTimeout(() => status.textContent = '', 2000);
}
